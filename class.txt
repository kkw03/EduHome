' ==========================================
' ENTITY CLASSES (Data Objects)
' ==========================================
package "Entity (Data Model)" {

    enum ZoneTier {
        GOLD_1KM
        SILVER_2KM
    }

    class User <<Entity>> {
        - userID: Int
        - email: String
        - passwordHash: String
        - contactNo: Int
        + authenticate(pw: String): Boolean
        + updateContact(email: String, no: Int): void
    }

    class PrimarySchool <<Entity>> {
        - schoolID: Int
        - officialName: String
        - postalCode: Int
        - latitude: Float
        - longitude: Float
        - vacancies: Int
        + getDetails(): String
        + updateVacancies(count: Int): void
    }

    class PriorityZone <<Entity>> {
        - zoneID: Int
        - radiusTier: ZoneTier
        - geometry: Shape
        + containsCoordinate(lat: Float, lng: Float): Boolean
    }

    class HDBBlock <<Entity>> {
        - blockID: Int
        - streetName: String
        - blockNum: String
        - leaseStartYear: Int
        - totalUnits: Int
        - latitude: Float
        - longitude: Float
        + getRemainingLease(currentYear: Int): Int
        + getLatestTransactions(): List<Transaction>
    }

    class Transaction <<Entity>> {
        - transactionID: Int
        - resalePrice: Float
        - floorAreaSqm: Int
        - transactionDate: Date
        - flatType: String
        + calculatePSF(): Float
    }

    class WatchlistItem <<Entity>> {
        - watchID: Int
        - minBudget: Float
        - maxBudget: Float
        - isActive: Boolean
        + updateBudget(min: Float, max: Float): void
    }

    class HDBBlockSet <<Entity>> {
        - blocks: List<HDBBlock>
        + addBlock(block: HDBBlock): void
        + removeBlock(blockID: Int): void
        + findByZone(zone: PriorityZone): List<HDBBlock>
    }

    class TransactionHistory <<Entity>> {
        - transactions: List<Transaction>
        + addTransaction(t: Transaction): void
        + getRecentTransactions(): List<Transaction>
    }

    class Watchlist <<Entity>> {
        - items: List<WatchlistItem>
        + addItem(item: WatchlistItem): void
        + removeItem(watchID: Int): void
        + getItems(): List<WatchlistItem>
    }

    class PersistentStore <<Entity>> {
        + saveTransaction(t: Transaction): void
        + loadTransactions(): List<Transaction>
        + saveWatchlist(w: Watchlist): void
        + loadWatchlist(userID: Int): Watchlist
        + saveUser(user: User): void
    }
}
